<script>
    const params = new URLSearchParams(window.location.search);
    const targetDateStr = params.get('date') || '2025-10-15T00:00:00';
    const targetDate = new Date(targetDateStr);
    const finalMessageText = params.get('message') || '–í—Ä–µ–º—è –≤—ã—à–ª–æ! üéâ';
    const bgColor = params.get('bgcolor'), blockColor = params.get('blockcolor'), numberColor = params.get('numbercolor'), labelColor = params.get('labelcolor');
    const root = document.documentElement;
    if (bgColor) root.style.setProperty('--bg-color', `#${bgColor}`);
    if (blockColor) root.style.setProperty('--block-color', `#${blockColor}`);
    if (numberColor) root.style.setProperty('--number-color', `#${numberColor}`);
    if (labelColor) root.style.setProperty('--label-color', `#${labelColor}`);
    const daysEl = document.getElementById("days"), hoursEl = document.getElementById("hours"), minutesEl = document.getElementById("minutes"), secondsEl = document.getElementById("seconds"), timerWrapper = document.getElementById("timer-wrapper"), finalMessage = document.getElementById("final-message");
    finalMessage.textContent = finalMessageText;
    const timer = setInterval(function() {
        const now = new Date().getTime(), diff = targetDate - now;
        if (diff <= 0) {
            clearInterval(timer);
            timerWrapper.style.display = 'none';
            finalMessage.style.display = 'block';
            return;
        }
        let days = Math.floor(diff / (1000 * 60 * 60 * 24)), hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)), minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60)), seconds = Math.floor((diff % (1000 * 60)) / 1000);
        daysEl.innerHTML = days;
        hoursEl.innerHTML = hours < 10 ? '0' + hours : hours;
        minutesEl.innerHTML = minutes < 10 ? '0' + minutes : minutes;
        secondsEl.innerHTML = seconds < 10 ? '0' + seconds : seconds;
    }, 1000);
</script>
